# backend

    ../docker/php/start shell    bash
    ../docker/php/start backend  PHP dev. server

    ../docker/phpcs              Run PHP Code Sniffer
    ../docker/phpmd              Run PHP Mess Detector

# frontend

    ../docker/js/start shell    bash
